Custom Unity Events, Custom Info

Unity already has a pretty good event system- UnityEvents can be registered in scripts to fire off to registered listeners, with specific numbers of arguments of almost any type. Since my Capstone game had a system of Instances to retain conceptual information when their corresponding objects weren't active in the scene, I wrote an extension that inherited from the original UnityEvent class and specified a series of identifiers that would be passed to listener functions, with some expanded functionality.

Unity Event Extended Class

[System.Serializable]
public class HonkEvent : UnityEvent
{
  //Properties
  protected HonkEventType eventType;
  protected string subjectName;
  protected HonkObject subjectObject = null;
  protected HonkEntity subjecyEntity = null;
  protected string message;

  //Getters
  public HonkEventType Type { get { return eventType; } }
  public string SubjectName { get { return subjectName; } }
  public HonkObject SubjectObject { get { return subjectObject; } }
  public HonkEntity SubjectEntity { get { return subjecyEntity; } }
  public string Message { get { return message; } }

  public virtual void InitHonkEvent(HonkEventType type, string name, HonkEntity entity = null)
  {
    eventType = type;
    subjectName = name;
    subjecyEntity = entity;
    if (subjecyEntity != null)
    {
      if (subjecyEntity is HonkInstance hi)
      {
        subjectObject = hi.honkObject;
      }
    }
    message = string.Empty;
  }
  public virtual void setMessage(string newMessage)
  {
    message = newMessage;
  }
}
